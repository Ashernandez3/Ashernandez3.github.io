DIY Photoshop

I created 10 different brushes in p5.js. Here is my design and the coding. Scroll down to try out the brushes.

var initials = 'ah'; 
var choice = '1';
var screenbg = 240;
var lastscreenshot = 61;
var colorPicker;

function setup() {
  createCanvas(600, 600);
  background(screenbg);
  cursor(CROSS);
  colorPicker = createColorPicker('#000000');
  colorPicker.position(10, 10);
}

function draw() {
  if (keyIsPressed) {
    choice = key;
    clear_print();
  }

  if (mouseIsPressed) {
    if (mouseButton == LEFT) {
      stroke(colorPicker.color());
      newkeyChoice(choice);
    } else if (mouseButton == RIGHT) {
      stroke(screenbg);
      strokeWeight(25);
      line(mouseX, mouseY, pmouseX, pmouseY);
    }
  }
}

function newkeyChoice(toolChoice) {
  if (toolChoice == '1') { strokeWeight(2); line(mouseX, mouseY, pmouseX, pmouseY); }
  else if (toolChoice == '2') { strokeWeight(25); line(mouseX, mouseY, pmouseX, pmouseY); }
  else if (toolChoice == '3') { stroke(screenbg); strokeWeight(10); line(mouseX, mouseY, pmouseX, pmouseY); }
  else if (toolChoice == '4') { stroke(screenbg); strokeWeight(60); line(mouseX, mouseY, pmouseX, pmouseY); }
  else if (toolChoice == '5') { noFill(); strokeWeight(5); for (let i=0;i<5;i++){ point(mouseX+random(-5,5),mouseY+random(-5,5)); } }
  else if (toolChoice == '6') { for (let i=0;i<40;i++){ point(mouseX+random(-15,15),mouseY+random(-15,15)); } }
  else if (toolChoice == '7') { noStroke(); fill(colorPicker.color()); square(mouseX,mouseY,15); }
  else if (toolChoice == '8') { for (let i=0;i<20;i++){ strokeWeight(random(1,4)); point(mouseX+random(-10,10),mouseY+random(-10,10)); } }
  else if (toolChoice == '9') { let c=colorPicker.color(); for (let i=0;i<10;i++){ stroke(red(c),green(c),blue(c),20*i); strokeWeight(i); line(mouseX,mouseY,pmouseX,pmouseY); } }
  else if (toolChoice == '0') { push(); translate(mouseX,mouseY); rotate(radians(45)); strokeWeight(10); line(0,0,pmouseX-mouseX,pmouseY-mouseY); pop(); }
}

function clear_print() {
  if (key == 'x' || key == 'X') { background(screenbg); }
  else if (key == 'p' || key == 'P') { saveme(); }
}

function saveme() {
  let filename = initials + day() + hour() + minute() + second();
  if (second() != lastscreenshot) { saveCanvas(filename, 'jpg'); key=""; }
  lastscreenshot = second();
}
